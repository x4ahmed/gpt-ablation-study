\begin{frame}{Evaluation Metric}
  \small
  The same validation calculation is applied to every model-weight candidate.

  \vspace{0.45cm}
  \textbf{1. Per-token negative log-likelihood}
  \[
    \mathrm{CE}_i(\theta_e)=-\log p_{\theta_e}(t_i\mid t_{<i})
  \]
  For each real token, measure how much probability the model assigned to the
  correct next token. A confident correct prediction produces a small loss.

  \vspace{0.4cm}
  \textbf{2. Padding-masked mean}
  \[
    m_i=\mathbb{1}[t_i\neq\mathrm{PAD}],
    \qquad
    \mathcal{L}_{\mathrm{val}}(\theta_e)=
    \frac{\sum_i \mathrm{CE}_i(\theta_e)m_i}{\sum_i m_i}
  \]
  Ignore padding positions, then average over the real validation tokens.
  Lower validation loss is better.

  \vspace{0.3cm}
  \begin{center}
    Only the evaluated weights change on the next slide; this loss calculation stays fixed.
  \end{center}
\end{frame}

\begin{frame}{Best Validation Loss}
  \small
  \textbf{1. Raw Val Loss}
  \[
    \mathcal{L}_{\mathrm{raw}}^{*}
      =\min_{1\le e\le E}\mathcal{L}_{\mathrm{val}}(\theta_e),
    \qquad E=16
  \]
  Evaluate raw weights after every epoch. Epoch 16 is the \emph{Final Val Loss};
  the candidate above is the best raw value from epochs 1--16.

  \vspace{0.1cm}
  \textbf{2. EMA Val Loss}
  \[
    \bar\theta_s=\beta\bar\theta_{s-1}+(1-\beta)\theta_s,
    \qquad s=1,\ldots,S
  \]
  Smooth the weights across optimizer steps, then evaluate $\bar\theta_S$ once after training.

  \vspace{0.1cm}
  \textbf{3. Last-Four Checkpoint Average}
  \[
    \theta_{\mathrm{avg}}
      =\frac{1\theta_{13}+2\theta_{14}+3\theta_{15}+4\theta_{16}}{10}
  \]
  Average epochs 13--16 with weights $1{:}2{:}3{:}4$, then evaluate once.

  \vspace{0.1cm}
  \begin{block}{Reported best validation loss}
    \centering
    \(\displaystyle
      \mathcal{L}_{\mathrm{best}}=\min\!\left\{
      \mathcal{L}_{\mathrm{raw}}^{*},
      \mathcal{L}_{\mathrm{val}}(\bar\theta_S),
      \mathcal{L}_{\mathrm{val}}(\theta_{\mathrm{avg}})
      \right\}
    \)
  \end{block}
\end{frame}

\begin{frame}{Architecture Changes}
  \footnotesize
  \textbf{Attention gate.}
  Baseline uses a sigmoid multiplier; strong gate doubles its range; no gate removes it.
  \makebox[\textwidth][c]{\(
    y_h'=\sigma(a_h(x))y_h
    \quad\longrightarrow\quad
    2\sigma(a_h(x))y_h
    \quad\text{or}\quad y_h
  \)}
  \vspace{0.08cm}\hrule\vspace{0.10cm}

  \textbf{Mirrored U-Net skips.}
  A decoder layer adds its matching encoder state; no-skip removes that term.
  \makebox[\textwidth][c]{\(
    x_\ell \leftarrow x_\ell+s_\ell x_{L-1-\ell}
    \quad\longrightarrow\quad x_\ell
  \)}
  \vspace{0.08cm}\hrule\vspace{0.10cm}

  \textbf{Value residuals.}
  Alternating layers inject a gated projection of $x_0$; no-value-residual keeps only $X_\ell W_\ell^V$.
  \makebox[\textwidth][c]{\(
    \widetilde V_\ell=X_\ell W_\ell^V
      +2\sigma(g_\ell(X_\ell))\odot P_\ell(x_0)
    \quad\longrightarrow\quad X_\ell W_\ell^V
  \)}
  \vspace{0.08cm}\hrule\vspace{0.10cm}

  \textbf{RoPE coverage.}
  Half RoPE rotates half of each $Q/K$ head; full RoPE rotates every head dimension.
  \makebox[\textwidth][c]{\(
    \operatorname{RoPE}_{1:d_h/2}(Q,K)
    \quad\longrightarrow\quad
    \operatorname{RoPE}_{1:d_h}(Q,K)
  \)}
  \vspace{0.08cm}\hrule\vspace{0.10cm}

  \textbf{Partial key offset.}
  Long-window and final layers shift the stationary key half by one token; no-offset leaves it unchanged.
  \makebox[\textwidth][c]{\(
    K_{t,\,d_h/2:d_h}\leftarrow K_{t-1,\,d_h/2:d_h}
    \quad\longrightarrow\quad K_t
  \)}
  \vspace{0.04cm}
  \smallcite{RoPE \cite{su2021roformer}; value residuals \cite{zhou2024resformer}}
\end{frame}

\begin{frame}{Effective Batch Size}
  \begin{center}
    \Large
    $16$ sequences/GPU $\times$ $2048$ tokens $\times$ $8$ GPUs\\[0.3em]
    $=262{,}144$ tokens per micro-step
  \end{center}
  \vspace{0.4cm}
  \begin{center}
    Two accumulated micro-steps\\[0.3em]
    $\Rightarrow 524{,}288$ tokens per optimizer update
  \end{center}
  \vspace{0.5cm}
  \begin{block}{Why this matters}
    Batch size and model width change the optimization regime. A hyperparameter
    that wins on the laptop is a hypothesis for scaling, not a guaranteed winner.
  \end{block}
\end{frame}

    \item Low WD clearly causes overfitting: final train loss falls to

  \end{itemize}

  \sectiondivider{Conclusion}
\end{frame}

\begin{frame}{Experiment Results}
  \begin{itemize}
    \item \textbf{Optimizer stability:} AdamW showed likely exploding gradients
      at LR 0.8; LR 0.6 was stable.
    \item \textbf{Overfitting:} WD 0.2 lowered training loss but worsened H100
      validation loss.
    \item \textbf{Compatibility:} low initial WD is incompatible with the
      baseline's late WD ramp and checkpoint averaging.
    \item \textbf{Reproducibility:} code, configurations, and result summaries
      are publicly available.\newline
      \href{https://github.com/x4ahmed/gpt-ablation-study}{\texttt{x4ahmed/gpt-ablation-study}}
    \item \textbf{Tiny Track reproduction:} H100 baseline 3.3343 vs. the
      clone-time record 3.332 (difference: 0.0023).
  \end{itemize}
\end{frame}

\begin{frame}{Limitations}
  \small
  \begin{tabular}{p{0.31\textwidth}p{0.61\textwidth}}
    \toprule
    Threat & Consequence \\
    \midrule
    One seed per setting & no empirical variance; sub-percent gaps may be noise \\
    Scale changes together & model, data, batch, hardware, and kernels are not isolated \\
    RoPE evidence & controlled no-offset comparison differs by only 0.0070;
      one seed cannot resolve it \\
    Value-residual ablation & removes both a path and 0.8\% of parameters \\
    Checkpoint selection & best across raw/EMA/average is optimistically selected \\
    \bottomrule
  \end{tabular}
  \par
  \vspace{0.4cm}
  These limitations narrow the claims; they do not erase the controlled
  strong-gate weight-decay reversal or the optimizer stability observation.
\end{frame}

\begin{frame}{Take-Home Message}
  \begin{enumerate}
    \item \textbf{Muon was the strongest optimizer tested.} It tolerated the
      aggressive matrix LR and beat stable AdamW at small scale.
    \item \textbf{Most architectural tweaks were minor.} Value residuals produced
      the largest observed architecture signal, with a parameter-count confound.
    \item \textbf{Small-scale tuning did not reliably transfer.} WD 0.2 improved
      laptop-scale loss by 6.3\% but degraded a controlled H100 pair by about 4\%.
    \item \textbf{Schedules must be studied as systems.} Weight decay, LR cycling,
      EMA, and checkpoint averaging interact late in training.
  \end{enumerate}
  \vspace{0.5cm}
  \begin{center}
    \Large Cheap ablations generate hypotheses; scaling experiments decide which survive.
  \end{center}
\end{frame}
